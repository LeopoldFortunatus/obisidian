# Как обновить сертфикаты

```
ssh -p 1985 root@new.catherine.spb.ru
docker stop catherinespbru_wordpress_1
docker stop catherinespbru_redirect_1
certbot certonly --force-renew -d catherine.spb.ru --server https://acme-v02.api.letsencrypt.org/directory
cd /var/lib/docker/volumes/catherinespbru_wordpress_data/_data/apache/conf/bitnami/certs
cp -v /etc/letsencrypt/archive/catherine.spb.ru/* ./
cp fullchain27.pem server.crt
cp privkey27.pem server.key
docker restart catherinespbru_wordpress_1
docker restart catherinespbru_redirect_1
```
